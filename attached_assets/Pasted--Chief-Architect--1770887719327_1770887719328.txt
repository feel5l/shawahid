Step 3: Inject Critical Auth Logic (Duplicate Prevention)
Open server/routes.ts. Locate the PATCH /api/user route (Teacher Onboarding). Replace the entire route handler with this robust logic:
app.patch("/api/user", async (req, res) => {
  if (!req.isAuthenticated()) return res.sendStatus(401);
  
  // Validate input
  const { mobileNumber, fullNameArabic, ...otherData } = req.body;
  if (!mobileNumber) return res.status(400).send("Mobile number is required");

  try {
    // 1. Check for Pre-existing User with this Mobile
    const existingUser = await db.query.users.findFirst({
      where: eq(users.mobileNumber, mobileNumber),
    });

    // 2. MERGE LOGIC: If user exists and is NOT the current session user
    if (existingUser && existingUser.id !== req.user.id) {
      console.log(`Merging session ${req.user.id} into existing user ${existingUser.id}`);
      
      // Update the OLD user with NEW details (if any)
      const [updatedUser] = await db
        .update(users)
        .set({ 
          fullNameArabic, 
          mobileNumber, 
          onboardingCompleted: true,
          ...otherData 
        })
        .where(eq(users.id, existingUser.id))
        .returning();

      // Delete the TEMPORARY user (the one created by Replit Auth just now)
      await db.delete(users).where(eq(users.id, req.user.id));

      // SWAP SESSION: Manually update Passport session to point to the OLD ID
      req.login(updatedUser, (err) => {
        if (err) return res.status(500).send("Session merge failed");
        return res.json(updatedUser);
      });
      return;
    }

    // 3. NORMAL UPDATE: No duplicate found
    const [updatedUser] = await db
      .update(users)
      .set({ 
        fullNameArabic, 
        mobileNumber, 
        onboardingCompleted: true,
        ...otherData 
      })
      .where(eq(users.id, req.user.id))
      .returning();

    res.json(updatedUser);
  } catch (error) {
    console.error("Onboarding Error:", error);
    res.status(500).send("Failed to update profile");
  }
});
