### Task 1: Update Server Routes
Modify `server/routes.ts` to:
1. Import `NotificationService` and `AuditService`.
2. Add new API endpoints for notifications:
   - `GET /api/notifications`: Get current user's notifications.
   - `POST /api/notifications/:id/mark-read`: Mark specific notification as read.
3. Update the Principal Approval/Rejection routes (`POST /api/principal/signatures/:id/approve` and `reject`) to:
   - Call `AuditService.log` to record the decision.
   - Call `NotificationService.send` to alert the teacher.

Use this code logic for the new routes:

```typescript
// Add these imports
import { NotificationService } from "./services/notification";
import { AuditService } from "./services/audit";

// Add these routes inside the setupAuth(app) block:

// --- Notification Routes ---
app.get("/api/notifications", async (req, res) => {
  if (!req.user) return res.sendStatus(401);
  const notes = await NotificationService.getUserNotifications(req.user.id);
  res.json(notes);
});

app.post("/api/notifications/:id/mark-read", async (req, res) => {
  if (!req.user) return res.sendStatus(401);
  await NotificationService.markAsRead(parseInt(req.params.id));
  res.sendStatus(200);
});

// --- Update Existing Approval Route (Replace logic) ---
app.post("/api/principal/signatures/:id/approve", isPrincipal, async (req, res) => {
  const signatureId = parseInt(req.params.id);
  const notes = req.body.notes;
  
  const signature = await storage.approveSignature(signatureId, notes);
  
  // Log & Notify
  await AuditService.log({
    userId: req.user!.id,
    action: "APPROVE",
    entityType: "signature",
    entityId: signatureId,
    details: { notes },
    ipAddress: req.ip
  });

  await NotificationService.send({
    recipientId: signature.userId,
    type: "success",
    title: "تم اعتماد الميثاق",
    message: "قام مدير المدرسة باعتماد ميثاق الأداء الخاص بك.",
    link: "/home"
  });

  res.json(signature);
});

// --- Update Existing Rejection Route (Replace logic) ---
app.post("/api/principal/signatures/:id/reject", isPrincipal, async (req, res) => {
  const signatureId = parseInt(req.params.id);
  const reason = req.body.reason;

  if (!reason) return res.status(400).send("Rejection reason required");

  const signature = await storage.rejectSignature(signatureId, reason);

  // Log & Notify
  await AuditService.log({
    userId: req.user!.id,
    action: "REJECT",
    entityType: "signature",
    entityId: signatureId,
    details: { reason },
    ipAddress: req.ip
  });

  await NotificationService.send({
    recipientId: signature.userId,
    type: "error",
    title: "تم رفض الميثاق",
    message: `تم إعادة الميثاق للمراجعة. السبب: ${reason}`,
    link: "/home"
  });

  res.json(signature);
});
