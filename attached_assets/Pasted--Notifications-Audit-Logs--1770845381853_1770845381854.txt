```typescript
import { stringify } from "csv-stringify/sync"; // You might need to install 'csv-stringify' if not present, or use manual string building for simplicity.

// Manual CSV builder to avoid new deps if possible
app.get("/api/principal/export-csv", isPrincipal, async (req, res) => {
  const teachers = await storage.getTeachers();
  // Flatten data for CSV
  const csvRows = [
    ["Job Number", "Name", "Specialization", "Indicators Count", "Pending Signatures"], // Header
    ...await Promise.all(teachers.map(async (t) => {
      const indicators = await storage.getUserIndicators(t.id);
      const signatures = await storage.getTeacherSignatures(t.id);
      const pending = signatures.filter(s => s.status === 'pending').length;
      return [t.jobNumber, t.fullNameArabic, t.specialization, indicators.length, pending];
    }))
  ];

  const csvContent = csvRows.map(e => e.join(",")).join("\n");
  
  res.header("Content-Type", "text/csv");
  res.header("Content-Disposition", "attachment; filename=school_performance_report.csv");
  res.send("\ufeff" + csvContent); // Add BOM for Excel Arabic support
});
